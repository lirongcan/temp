class CommonData{
    private volatile static boolean flag=false;
    private volatile static int i=1;
    private volatile static int k=100;

    void printThread1(){
        try{
            while(k>0){
                synchronized (this){
                    if(flag){
                        this.wait();
                    }
                    else{
                        this.notify();
                        System.out.println(Thread.currentThread().getName()+": "+String.valueOf(i));
                        i++;
                        flag=true;
                        k--;
                    }
                }
            }
        }
        catch (InterruptedException e){
            e.printStackTrace();
        }
    }


    void printThread2(){
        try{
            while(k>0){
                synchronized (this){
                    if(!flag){
                        this.wait();
                    }
                    else{
                        this.notify();
                        System.out.println(Thread.currentThread().getName()+": "+String.valueOf(i));
                        i++;
                        flag=false;
                        k--;
                    }
                }
            }
        }
        catch (InterruptedException e){
            e.printStackTrace();
        }
    }
}
class Thread1 extends Thread{
    private final CommonData data;
    Thread1(CommonData data){
        this.data=data;
    }

    @Override
    public void run() {
        data.printThread1();
    }
}

class Thread2 extends Thread{
    private CommonData data;
    Thread2(CommonData data){
        this.data=data;
    }

    @Override
    public void run() {
        data.printThread2();
    }
}

public class TwoThreadPrintCommonData {
    public void test(){
        CommonData data=new CommonData();
        new Thread1(data).start();
        new Thread2(data).start();
    }
}
