//topk问题啊，直接用堆




//给定数组，每个元素表示一个木头的长度，木头可以截断，从这堆木头中截出至少k个相同长度为m的木块
//已知k，求m
class Solution {
    int MaxM(int[] nums, int k) {
        int min=Integer.MAX_VALUE;
        int max=Integer.MIN_VALUE;
        for(int a:nums){
            min=Math.min(min,a);
            max=Math.max(max,a);
        }
        int l=min,r=max;
        int ans=-1;
        while(l<=r){
            int mid=(l+r)/2;
            if(isFit(nums,k,mid)){
                ans=mid;
                l=mid+1;
            }
            else r=mid-1;
        }
        return ans;
    }
    boolean isFit(int[]nums,int k,int mid){
        int cnt=0;
        for (int a : nums) {
            cnt += a / mid;
        }
        return cnt>=k;
    }
}
